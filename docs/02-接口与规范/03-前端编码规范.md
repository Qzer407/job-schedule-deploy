## 三、TypeScript 规范

### 3.1 类型定义

所有接口和类型都必须正确定义，使用 `interface` 或 `type`。

```typescript
// types/user.ts
export interface User {
  id: number
  username: string
  email?: string
  role: string
  status: number
}

export interface LoginRequest {
  username: string
  password: string
}
```

### 3.2 组件类型

Vue 组件必须使用 `<script setup lang="ts">` 语法。

```vue
<script setup lang="ts">
import { ref, reactive } from 'vue'
import type { User } from '@/types'

const user = ref<User | null>(null)
const form = reactive<User>({
  id: 0,
  username: '',
  role: 'USER',
  status: 1
})
</script>
```

### 3.3 API 返回类型

API 请求必须定义正确的返回类型。

```typescript
import type { ApiResponse, PageResult } from '@/types'

export const getUser = (id: number) => {
  return request.get<ApiResponse<User>>(`/api/v1/users/${id}`)
}

export const getUserList = (params: PageParams) => {
  return request.get<ApiResponse<PageResult<User>>>('/api/v1/users', { params })
}
```

## 四、命名规范

| 类型 | 命名规则 | 示例 |
|-----|---------|------|
| 组件文件名 | PascalCase | `TaskManager.vue` |
| 组件名 | PascalCase | `<TaskManager />` |
| 类型/接口 | PascalCase | `UserInfo`, `LoginRequest` |
| 方法名 | camelCase | `handleCreate`, `loadData` |
| 变量名 | camelCase | `taskList`, `searchParams` |
| 常量 | UPPER_SNAKE_CASE | `DEFAULT_PAGE_SIZE` |
| API方法名 | camelCase | `getTaskList`, `createTask` |
| 类型文件 | kebab-case | `user-info.ts` |

## 五、代码规范

### 5.1 Vue 组件规范

#### 5.1.1 组件结构

```vue
<script setup lang="ts">
import { ref, reactive, onMounted } from 'vue'
import { ElMessage } from 'element-plus'
import { getTaskList, createTask } from '@/api'
import type { TaskInfo, TaskCreateRequest } from '@/types'

const tableData = ref<TaskInfo[]>([])
const loading = ref(false)

const loadData = async () => {
  loading.value = true
  try {
    const res = await getTaskList()
    tableData.value = res.data.data.records
  } catch (e) {
    console.error(e)
  } finally {
    loading.value = false
  }
}

onMounted(() => {
  loadData()
})
</script>

<template>
  <div class="task-manager">
    <el-table :data="tableData" v-loading="loading">
      <!-- 表格内容 -->
    </el-table>
  </div>
</template>

<style scoped>
.task-manager {
  padding: 20px;
}
</style>
```

### 5.2 API 请求规范

```typescript
// api/index.ts
import axios from 'axios'
import type { ApiResponse } from '@/types'

const request = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || 'http://localhost:8080',
  timeout: 10000
})

request.interceptors.request.use(config => {
  const token = localStorage.getItem('token')
  if (token) {
    config.headers['Authorization'] = `Bearer ${token}`
  }
  return config
})

export const getTaskList = (params?: PageParams) => {
  return request.get<ApiResponse<PageResult<TaskInfo>>>('/api/v1/tasks', { params })
}
```

## 八、路由规范

### 8.1 路由配置

```typescript
// main.ts
import type { RouteRecordRaw } from 'vue-router'

const routes: RouteRecordRaw[] = [
  { path: '/login', component: Login },
  { path: '/', redirect: '/task' },
  { path: '/task', component: TaskManager, meta: { requiresAuth: true } }
]
```

### 8.2 路由守卫

```typescript
router.beforeEach((to, from, next) => {
  const token = localStorage.getItem('token')
  if (to.meta.requiresAuth && !token) {
    next('/login')
  } else {
    next()
  }
})
```
